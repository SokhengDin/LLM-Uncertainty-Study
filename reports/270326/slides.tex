\begin{frame}{Objective}

\begin{block}{\textbf{Problem}}
  Current LLM leaderboards (HuggingFace, OpenCompass \ldots) rank models
  by \textcolor{darkred}{\textbf{accuracy only}}.\\[5pt]
  Two models with \emph{equal accuracy} may differ greatly in
  \textbf{uncertainty} -- yet benchmarks treat them identically.
\end{block}

\bigskip

\begin{block}{\textbf{Goal}}
  Integrate \textbf{uncertainty quantification} into LLM benchmarking
  via \textbf{conformal prediction} --
  a statistically rigorous, distribution-free, model-agnostic method.
\end{block}

\end{frame}

% ================================================================
\begin{frame}{Methodology -- Conformal Prediction}

\begin{block}{\textbf{Coverage Guarantee}}
  For test instance $X_t$ with true label $Y_t$, CP outputs a
  \textbf{prediction set} $\mathcal{C}(X_t)\subseteq\mathcal{Y}$ such that:
  $$P\!\bigl(Y_t \in \mathcal{C}(X_t)\bigr) \;\geq\; 1-\alpha$$
  $\alpha$ = user-specified error rate \quad
  (e.g.\ $\alpha=0.1 \;\Rightarrow\;$ \textbf{90\,\%} coverage guarantee).
\end{block}

\bigskip

\begin{block}{\textbf{Uncertainty Measure}}
  \textbf{Set Size (SS)} $=$ average $|\mathcal{C}(X_t)|$ over all test instances.\\[4pt]
  \hspace*{1em}
  \textcolor{steelblue}{Small SS} $\Rightarrow$ \textbf{confident model} \quad
  \textcolor{darkred}{Large SS} $\Rightarrow$ \textbf{uncertain model}
\end{block}

\end{frame}

% ================================================================
\begin{frame}{Methodology -- Score Functions}

\begin{columns}[T]
  \begin{column}{0.48\textwidth}
    \begin{block}{\textbf{LAC} \small-- Least Ambiguous Classifier}
      $$s(X,Y)=1-f(X)_Y$$
      Uses softmax score of the true label only.\\[4pt]
      \textcolor{steelblue}{$\Rightarrow$ tends to smaller sets.}
    \end{block}
  \end{column}
  \begin{column}{0.48\textwidth}
    \begin{block}{\textbf{APS} \small-- Adaptive Prediction Sets}
      $$s(X,Y)=\!\sum_{\substack{Y'\in\mathcal{Y}:\\f(X)_{Y'}\geq f(X)_Y}}\!f(X)_{Y'}$$
      Accumulates ranked softmax scores.\\[4pt]
      \textcolor{steelblue}{$\Rightarrow$ better on hard instances.}
    \end{block}
  \end{column}
\end{columns}

\begin{frame}{Algorithm -- CP for LLMs}

\begin{tcolorbox}[
  enhanced, skin=bicolor,
  colback=codebg, colbacklower=codebg!60,
  colframe=steelblue!60, arc=3pt,
  boxrule=0.6pt, left=4pt, right=4pt, top=3pt, bottom=3pt,
  title={\scriptsize\textbf{Algorithm 1:} Conformal Prediction for LLMs
         \hfill\textcolor{white!70}{\scriptsize
           \textbf{Input:} $\mathcal{M},\,\mathcal{D},\,\alpha$
           \quad\textbf{Output:} Acc,\,SS,\,CR,\,UAcc}},
  fonttitle=\small\bfseries\color{white},
  coltitle=white, attach boxed title to top left,
  boxed title style={colback=steelblue!80, arc=2pt, boxrule=0pt,
                     left=4pt, right=4pt}
]
\scriptsize
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{@{\,}r@{\;\;}l@{\quad}l@{}}
  \textcolor{steelblue}{\textbf{1}} &
    \pckw{for each} prompt strategy $p \in \{$base, shared, task$\}$ \pckw{do} &
    \pccomment{3 strategies} \\
  \textcolor{steelblue}{\textbf{2}} &
    \quad\pckw{for each} ICL setting $k \in \{$0-shot, few-shot, CoT$\}$ \pckw{do} &
    \pccomment{3 ICL modes} \\
  \textcolor{steelblue}{\textbf{3}} &
    \quad\quad $P(X)\leftarrow\mathrm{Softmax}(\mathrm{logits}_{\mathcal{M}}(X_{p,k}))$\; \pckw{for all} $X\in\mathcal{D}$ &
    \pccomment{query LLM} \\
  \textcolor{steelblue}{\textbf{4}} &
    \quad\quad $\mathcal{D}\to\mathcal{D}_{\mathrm{cal}}\ (50\%)+\mathcal{D}_{\mathrm{test}}\ (50\%)$ &
    \pccomment{random split} \\
  \textcolor{steelblue}{\textbf{5}} &
    \quad\quad \pckw{for each} $s \in \{$LAC, APS$\}$ \pckw{do} &
    \pccomment{2 score fns} \\
  \textcolor{steelblue}{\textbf{6}} &
    \quad\quad\quad $s_i \leftarrow s(X_c^{(i)}, Y_c^{(i)})$\; \pckw{for all} $i\in\mathcal{D}_{\mathrm{cal}}$ &
    \pccomment{nonconformity scores} \\
  \textcolor{steelblue}{\textbf{7}} &
    \quad\quad\quad $\hat{q}\leftarrow\mathrm{Quantile}\!\left(\{s_i\},\,\tfrac{\lceil(n+1)(1-\alpha)\rceil}{n}\right)$ &
    \pccomment{adaptive threshold} \\
  \textcolor{steelblue}{\textbf{8}} &
    \quad\quad\quad $\mathcal{C}(X_t)\leftarrow\{Y'\in\mathcal{Y}:s(X_t,Y')\leq\hat{q}\}$\; \pckw{for all} $X_t\in\mathcal{D}_{\mathrm{test}}$ &
    \pccomment{prediction sets} \\
  \textcolor{steelblue}{\textbf{9}} &
    \quad\quad\quad \pckw{if} $\mathcal{C}(X_t)=\emptyset$\;\pckw{then keep} $\arg\max_{Y'}P(X_t)_{Y'}$ & \\[2pt]
  \textcolor{steelblue}{\textbf{10}} &
    \pckw{return} avg over $p,k,s$:\;
    $\mathrm{Acc},\;\mathrm{SS},\;\mathrm{CR},\;
    \mathrm{UAcc}=\tfrac{\mathrm{Acc}}{\mathrm{SS}}\sqrt{|\mathcal{Y}|}$ &
\end{tabular}
\end{tcolorbox}

\end{frame}

% ================================================================
\begin{frame}{Methodology -- Evaluation Metrics}

\renewcommand{\arraystretch}{1.65}
\begin{tabular}{@{}lll@{}}
  \toprule
  \textbf{Metric} & \textbf{Formula} & \textbf{Meaning} \\
  \midrule
  Acc &
    $\dfrac{1}{|\mathcal{D}_{\mathrm{test}}|}\sum\mathbf{1}[Y_p=Y_t]$ &
    Prediction accuracy \\
  SS  &
    $\dfrac{1}{|\mathcal{D}_{\mathrm{test}}|}\sum|\mathcal{C}(X_t)|$ &
    Uncertainty ($\uparrow$ more uncertain) \\
  CR  &
    $\dfrac{1}{|\mathcal{D}_{\mathrm{test}}|}\sum\mathbf{1}[Y_t\in\mathcal{C}(X_t)]$ &
    Coverage ($\geq1{-}\alpha$ guaranteed) \\
  \bottomrule
\end{tabular}

\bigskip

\begin{block}{\textbf{New: Uncertainty-aware Accuracy (UAcc)}}
  $$\mathrm{UAcc}=\frac{\mathrm{Acc}}{\mathrm{SS}}\sqrt{|\mathcal{Y}|},
  \qquad |\mathcal{Y}|=6$$
  Rewards models that are both \textbf{accurate} and \textbf{certain}.
  Can \textbf{reverse rankings} vs.\ Acc alone.
\end{block}

\end{frame}

\begin{frame}{Results -- Three Key Findings}

\begin{block}{\textcolor{white}{\textbf{Finding 1:} Higher Accuracy $\neq$ Lower Uncertainty}}
  InternLM-7B outperforms MPT-7B by \textbf{+19.34\,\%} Acc on DRS,
  yet shows \textbf{higher} SS (3.63 vs.\ 3.59).
  Rankings by Acc and SS diverge across \emph{all} tasks.
\end{block}

\medskip

\begin{block}{\textcolor{white}{\textbf{Finding 2:} Larger $\neq$ More Certain}}
  Qwen 1.8B$\to$14B: Acc and certainty both improve.\\
  Qwen 14B$\to$72B: Acc rises, but SS \textbf{increases} on RC \& DRS -- non-monotonic.
\end{block}

\medskip

\begin{block}{\textcolor{white}{\textbf{Finding 3:} Instruction Finetuning Raises Uncertainty}}
  Chat-V1 and Chat-V2 show \textbf{consistently higher SS} than base models
  across Llama-2, Yi, DeepSeek, and Falcon series.
\end{block}

\end{frame}

\begin{frame}{Conclusion \& Future Work}

\begin{block}{\textbf{Conclusion}}
\begin{itemize}\setlength\itemsep{4pt}
  \item Accuracy alone is \textbf{insufficient} for LLM evaluation.
  \item Conformal prediction offers a \textbf{rigorous, model-agnostic} uncertainty measure.
  \item New \textbf{UAcc} metric jointly captures accuracy and certainty.
  \item Three findings: Acc\,$\neq$\,certainty;\; larger\,$\neq$\,more certain;\;
        finetuning raises uncertainty.
\end{itemize}
\end{block}

\medskip

\begin{block}{\textbf{Future Work}}
\begin{itemize}\setlength\itemsep{4pt}
  \item Extend CP to \textbf{free-form text generation} beyond MCQA.
  \item Apply to \textbf{multi-modal} and \textbf{closed-source} LLMs (GPT-4 etc.).
  \item Integrate uncertainty into \textbf{RAG} and \textbf{in-context learning} pipelines.
\end{itemize}
\end{block}
